final disableScrollProvider = StateProvider<bool>(
  (ref) {
    ref.listenSelf((previous, next) async {
      final prefs = await ref.watch(prefsProvider.future);
      await prefs.setBool('disableScroll', next);
    });
    return false;
  },
);
